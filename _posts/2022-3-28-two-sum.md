```python
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        # method 1
        dic = collections.defaultdict(int)
        for i in range(len(nums)):
            dic[nums[i]] = i
        for j in range(len(nums)):
            new = target - nums[j] 
            if new in dic and dic[new] != j:
                return [j, dic[new]]
        return None
```
